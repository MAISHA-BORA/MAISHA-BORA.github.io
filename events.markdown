<script>
// Simple JavaScript for event interactions
document.addEventListener('DOMContentLoaded', function() {
    // Smooth scrolling for navigation
    const navLinks = document.querySelectorAll('.events-navigation .nav-item');
    
    navLinks.forEach(link => {
        link.addEventListener('click', function(e) {
            e.preventDefault();
            const targetId = this.getAttribute('href');
            const targetElement = document.querySelector(targetId);
            
            if (targetElement) {
                targetElement.scrollIntoView({
                    behavior: 'smooth',
                    block: 'start'
                });
            }
        });
    });
    
    // Event registration buttons
    const registerButtons = document.querySelectorAll('.btn-event, .btn-training');
    
    registerButtons.forEach(button => {
        button.addEventListener('click', function(e) {
            if (this.getAttribute('href') === '#register') {
                e.preventDefault();
                document.getElementById('register').scrollIntoView({
                    behavior: 'smooth',
                    block: 'start'
                });
            }
        });
    });
    
    // Notify me buttons for TBA events
    const notifyButtons = document.querySelectorAll('a[href="#notify"]');
    
    notifyButtons.forEach(button => {
        button.addEventListener('click', function(e) {
            e.preventDefault();
            const eventTitle = this.closest('.event-actions').previousElementSibling.querySelector('.event-title').textContent;
            alert(`We'll notify you when registration opens for: ${eventTitle}`);
        });
    });
});
</script>
